ngx-code-highlight


Installation

To install this library, run:

npm install @ngx-tiny/code-highlight prismjs --save

and then add NgxCodeHighlightModule to you AppModule

Usage

Example Output:

Place the code between <![CDATA[]]> to ensure formatting and escaping of characters. 

      
<ngx-code-example>
  <![CDATA[
  import {{'{'}} NgModule {{'}'}} from '@angular/core';
  import {{'{'}} BrowserModule {{'}'}} from '@angular/platform-browser';
  import {{'{'}} NgxCodeHighlightModule {{'}'}} from '@ngx-tiny/code-highlight';
  
  import {{'{'}} AppComponent {{'}'}} from './app.component';
  
  @NgModule({{'{'}}
    imports: [BrowserModule, NgxCodeHighlightModule],
    declarations: [AppComponent],
    bootstrap: [AppComponent]
  {{'}'}})
  export class AppModule {{'{'}}
  ]]>
</ngx-code-example>

To ensure angular keeps the white spaces, add preserveWhitespaces : true to @Component.
